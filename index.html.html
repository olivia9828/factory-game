// proxy.js — node proxy.js 실행 const http=require('http'),https=require('https'); const API='https://testnet-api.fortem.gg'; http.createServer(async(req,res)=>{ res.setHeader('Access-Control-Allow-Origin','*'); res.setHeader('Access-Control-Allow-Headers','*'); if(req.method==='OPTIONS'){res.writeHead(200);res.end();return;} let body='';req.on('data',d=>body+=d); req.on('end',async()=>{ try{ const{apiKey,wallet,items,gameState}=JSON.parse(body); // Get auth token const tok=await new Promise((ok,err)=>{ const r=https.request(API+'/auth/nonce',{headers:{'x-api-key':apiKey}},r=>{ let d='';r.on('data',c=>d+=c);r.on('end',()=>{ const{nonce}=JSON.parse(d); https.request(API+'/auth/access-token',{method:'POST', headers:{'Content-Type':'application/json','x-api-key':apiKey}},r2=>{ let d2='';r2.on('data',c=>d2+=c); r2.on('end',()=>ok(JSON.parse(d2).token)); }).end(JSON.stringify({nonce})); }); });r.end(); }); // Mint items let minted=0; for(const item of items) for(let i=0;i<item.count;i++){ await new Promise(ok=>{ const b=JSON.stringify({name:item.name,walletAddress:wallet, attributes:[{trait:'Game',value:'Factory Rush'},{trait:'Type',value:item.name}]}); const r=https.request(API+'/collections/factory-rush/items', {method:'POST',headers:{'Content-Type':'application/json','Authorization':'Bearer '+tok}}, rr=>{let d='';rr.on('data',c=>d+=c);rr.on('end',()=>{minted++;ok();})}); r.write(b);r.end(); }); } res.writeHead(200,{'Content-Type':'application/json'}); res.end(JSON.stringify({ok:true,minted, mintedById:Object.fromEntries(items.map(i=>[i.id,i.count]))})); }catch(e){res.writeHead(500);res.end(JSON.stringify({ok:false,error:String(e)}));} }); }).listen(8787,()=>console.log('✓ Proxy at http://localhost:8787'));
